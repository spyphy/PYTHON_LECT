\section{Start}$~$
\url{http://pybrain.org/docs/tutorial/netmodcon.html}

\begin{lstlisting}[caption="Python"]
from pybrain.structure import RecurrentNetwork
net = RecurrentNetwork()

from pybrain.structure import LinearLayer, SigmoidLayer, FullConnection
net.addInputModule(LinearLayer(2, name='in'))
net.addModule(SigmoidLayer(3, name='hidden'))
net.addOutputModule(LinearLayer(1, name='out'))
net.addConnection(FullConnection(net['in'], net['hidden'], name='c1'))
net.addConnection(FullConnection(net['hidden'], net['out'], name='c2'))
net.addRecurrentConnection(FullConnection(net['hidden'], net['hidden'], name='c3'))
net.sortModules()

for i in range(0,3):
	print('calc:' + str(net.activate((2, 2))))
\end{lstlisting} 


\subsection{Add DataSet}$~$

\begin{lstlisting}[caption="Python"]
from pybrain.datasets import SupervisedDataSet
ds = SupervisedDataSet(2,1)
ds.addSample((0,0), (0,))
ds.addSample((0,1), (1,))
ds.addSample((1,0), (1,))
ds.addSample((1,1), (1,))
print(len(ds))
\end{lstlisting}

from pybrain.supervised.trainers import BackpropTrainer
